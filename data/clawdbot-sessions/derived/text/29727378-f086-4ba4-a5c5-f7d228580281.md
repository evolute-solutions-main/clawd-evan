"""Pull sales-side data for a date range (inclusive) and aggregate a weekly report.

import json
import os

from zoneinfo import ZoneInfo

def load_json(path: str):
    with open(path, "r", encoding="utf-8") as f:
        return json.load(f)


def to_dt(v, tz: ZoneInfo):
    if isinstance(v, (int, float)):
        if v > 1e12:
            v /= 1000
        return dt.datetime.fromtimestamp(v, tz=tz)
    if isinstance(v, str):
        try:
            x = dt.datetime.fromisoformat(v)
            if x.tzinfo is None:
                x = x.replace(tzinfo=tz)
            return x.astimezone(tz)
        except Exception:
            return None
    return None


def unwrap_list(body):
    if isinstance(body, list):
        return body
    if isinstance(body, dict):
        for k in ("items", "data", "results", "calendars", "events", "pipelines", "opportunities", "conversations", "users", "contacts"):
            v = body.get(k)
            if isinstance(v, list):
                return v
    return None


def extract_events(day_dir: str):
    p = os.path.join(day_dir, "calendarEvents.json")
    if not os.path.exists(p):
        return []
    obj = load_json(p)
    out = []
    by_cal = obj.get("byCalendar", {}) if isinstance(obj, dict) else {}
    for cal_id, info in by_cal.items():
        name = (info or {}).get("name")
        payload = (info or {}).get("events")
        items = unwrap_list(payload) or []
        for ev in items:
            if isinstance(ev, dict):
                ev2 = dict(ev)
                ev2["_calendarId"] = cal_id
                ev2["_calendarName"] = name
                out.append(ev2)
    return out


def extract_conversations_in_window(day_dir: str, start: dt.datetime, end: dt.datetime, tz: ZoneInfo):
    p = os.path.join(day_dir, "conversations.json")
    if not os.path.exists(p):
        return []
    payloads = load_json(p)
    if not isinstance(payloads, list) or not payloads:
        return []
    # pick the payload with the largest list (usually the limit=100 one)
    pl = max(payloads, key=lambda x: len((x or {}).get("conversations", [])) if isinstance(x, dict) else 0)
    convs = (pl or {}).get("conversations", []) if isinstance(pl, dict) else []
    inwin = []
    for c in convs:
        if not isinstance(c, dict):
            continue
        v = c.get("lastMessageDate") or c.get("lastMessageAt") or c.get("updatedAt") or c.get("dateUpdated")
        d = to_dt(v, tz)
        if d and start <= d < end:
            inwin.append(c)
    return inwin

        weekly["days"].append(
            {
                "date": ds,

                "appointments_by_calendar": dict(sorted(by_cal.items(), key=lambda x: (-x[1], x[0]))),

#!/usr/bin/env python3
"""Pull ALL conversations in a time window by iteratively advancing startDate.

Reason: `/conversations/search` in this account ignores pagination params (offset/page/etc),
but it DOES honor startDate/endDate and sort.

Strategy:
- Query with sort=asc, limit=100
- After each page, set new startDate = lastMessageDate(last item) + 1
- Stop when empty or no progress.

Outputs JSON list to stdout or --out file.
"""

from __future__ import annotations

import argparse
import datetime as dt
import json
import os
import sys
from zoneinfo import ZoneInfo

import requests

BASE = os.getenv("GHL_BASE_URL", "https://services.leadconnectorhq.com")
VER = "2021-07-28"


def req(session: requests.Session, params: dict):
    r = session.get(f"{BASE}/conversations/search", params=params, timeout=60)
    try:
        body = r.json()
    except Exception:
        body = r.text
    return r.status_code, body


def to_ms(x: dt.datetime) -> int:
    return int(x.timestamp() * 1000)


def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--tz", default="America/Bogota")
    ap.add_argument("--start", required=True, help="YYYY-MM-DDTHH:MM:SS or YYYY-MM-DD")
    ap.add_argument("--end", required=True, help="YYYY-MM-DDTHH:MM:SS or YYYY-MM-DD")
    ap.add_argument("--limit", type=int, default=100)
    ap.add_argument("--out", default="")
    args = ap.parse_args()

    tok = os.getenv("GHL_PRIVATE_INTEGRATION_TOKEN", "").strip()
    loc = os.getenv("GHL_LOCATION_ID", "").strip()
    if not tok or not loc:
        print("Missing GHL_PRIVATE_INTEGRATION_TOKEN or GHL_LOCATION_ID", file=sys.stderr)
        sys.exit(2)

    tz = ZoneInfo(args.tz)

    def parse(s: str) -> dt.datetime:
        if len(s) == 10:
            d = dt.date.fromisoformat(s)
            return dt.datetime(d.year, d.month, d.day, 0, 0, 0, tzinfo=tz)
        x = dt.datetime.fromisoformat(s)
        if x.tzinfo is None:
            x = x.replace(tzinfo=tz)
        return x.astimezone(tz)

    start = parse(args.start)
    end = parse(args.end)

    session = requests.Session()
    session.headers.update({
        "Authorization": f"Bearer {tok}",
        "Accept": "application/json",
        "Content-Type": "application/json",
        "Version": VER,
    })

    start_ms = to_ms(start)
    end_ms = to_ms(end)

    out = []
    seen_ids = set()

    guard = 0
    while True:
        guard += 1
        if guard > 5000:
            break

        params = {
            "locationId": loc,
            "startDate": start_ms,
            "endDate": end_ms,
            "limit": args.limit,
            "sort": "asc",
        }
        code, body = req(session, params)
        if code != 200:
            raise SystemExit(f"/conversations/search failed: {code} {body}")

        convs = body.get("conversations", []) if isinstance(body, dict) else []
        if not convs:
            break

        progressed = False
        last_msg = None

        for c in convs:
            cid = c.get("id")
            if cid and cid not in seen_ids:
                out.append(c)
                seen_ids.add(cid)
                progressed = True
            # track lastMessageDate for cursoring
            v = c.get("lastMessageDate")
            if isinstance(v, (int, float)):
                last_msg = int(v)

        if last_msg is None:
            # if no timestamp, cannot advance safely
            break

        # advance start date (exclusive)
        new_start = last_msg + 1
        if new_start <= start_ms:
            break
        start_ms = new_start

        if not progressed:
            # we didn't add anything new; avoid infinite loop
            break

        # if fewer than limit, likely done
        if len(convs) < args.limit:
            break

    obj = {
        "window": {"start": start.isoformat(), "end": end.isoformat(), "tz": args.tz},
        "locationId": loc,
        "count": len(out),
        "conversations": out,
    }

    if args.out:
        os.makedirs(os.path.dirname(args.out), exist_ok=True)
        with open(args.out, "w", encoding="utf-8") as f:
            json.dump(obj, f, indent=2)
        print(f"Wrote {args.out}")
    else:
        print(json.dumps(obj, indent=2))


if __name__ == "__main__":
    main()
