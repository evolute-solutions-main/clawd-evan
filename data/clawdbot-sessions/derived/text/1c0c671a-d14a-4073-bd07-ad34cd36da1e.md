## Quick web version (shuffled playlist, plays through all clips)
Here’s a complete single-file example. Replace the `clips` paths with your audio files.

**index.html**
```html
<!doctype html>
<html>
  <head>
    <meta charset="utf-8" />
    <title>Random Clip Player</title>
    <style>
      body { font-family: system-ui; padding: 20px; }
      button { padding: 10px 14px; margin-right: 8px; }
      #now { margin-top: 14px; }
      ul { margin-top: 10px; }
    </style>
  </head>
  <body>
    <h1>Random Clip Player</h1>

    <button id="play">Play (shuffle & start)</button>
    <button id="next">Next</button>
    <button id="stop">Stop</button>

    <div id="now">Now playing: —</div>
    <ul id="list"></ul>

    <audio id="audio"></audio>

    <script>
      // Put your clips here (relative paths or URLs)
      const clips = [
        "audio/clip1.mp3",
        "audio/clip2.mp3",
        "audio/clip3.mp3"
      ];

      const audio = document.getElementById("audio");
      const now = document.getElementById("now");
      const listEl = document.getElementById("list");

      // Render list
      clips.forEach(src => {
        const li = document.createElement("li");
        li.textContent = src;
        listEl.appendChild(li);
      });

      let queue = [];
      let idx = -1;

      function shuffle(arr) {
        // Fisher–Yates
        for (let i = arr.length - 1; i > 0; i--) {
          const j = Math.floor(Math.random() * (i + 1));
          [arr[i], arr[j]] = [arr[j], arr[i]];
        }
        return arr;
      }

      function startShuffled() {
        queue = shuffle([...clips]);
        idx = 0;
        playCurrent();
      }

      function playCurrent() {
        if (idx < 0 || idx >= queue.length) return;
        const src = queue[idx];
        audio.src = src;
        now.textContent = `Now playing: ${src} (${idx + 1}/${queue.length})`;
        audio.play();
      }

      function next() {
        if (!queue.length) startShuffled();
        else {
          idx++;
          if (idx >= queue.length) {
            // reshuffle and start again (or stop—your choice)
            startShuffled();
            return;
          }
          playCurrent();
        }
      }

      document.getElementById("play").onclick = startShuffled;
      document.getElementById("next").onclick = next;
      document.getElementById("stop").onclick = () => {
        audio.pause();
        audio.currentTime = 0;
        now.textContent = "Now playing: —";
      };

      audio.addEventListener("ended", next);
    </script>
  </body>
</html>
```

```html
      // Fisher–Yates shuffle
      for (let i = arr.length - 1; i > 0; i--) {

        [arr[i], arr[j]] = [arr[j], arr[i]];
      }
      return arr;
    }

    function setNowPlaying(text) {
      nowEl.innerHTML = `Now playing: <span>${text}</span>`;
    }

    function setStatus(text) {
      statusEl.textContent = text;
    }

    function startNewRun() {
      if (!clips.length) {
        setNowPlaying("—");
        setStatus("No clips configured.");
        return;

      index = 0;
      playCurrent();

      const src = queue[index];
      audio.src = src;
      setNowPlaying(`${src} (${index + 1}/${queue.length})`);
      setStatus("Playing…");
      audio.play();
    }

    function advance() {
      index += 1;

      if (index >= queue.length) {
        // Finished the full shuffled run (played each exactly once)
        audio.removeAttribute("src");
        audio.load();
        setNowPlaying("—");
        setStatus("Done. Click Play to shuffle and run again.");
        queue = [];
        index = -1;
        return;
      }

      playCurrent();
    }

    document.getElementById("play").addEventListener("click", startNewRun);

    document.getElementById("pause").addEventListener("click", () => {
      audio.pause();
      if (audio.currentTime > 0) setStatus("Paused.");
    });

    document.getElementById("resume").addEventListener("click", () => {
      if (audio.src) {
        audio.play();
        setStatus("Playing…");
      }
    });

    document.getElementById("stop").addEventListener("click", () => {
      audio.pause();
      audio.currentTime = 0;
      audio.removeAttribute("src");
      audio.load();
      queue = [];
      index = -1;
      setNowPlaying("—");
      setStatus("Stopped.");
    });

    audio.addEventListener("ended", advance);

    audio.addEventListener("error", () => {
      setStatus("Error loading/playing audio. Check file paths and server.");
    });
  </script>
</body>
```

## A) Pure frontend (user picks a folder)
This loads every audio file in a chosen folder (and subfolders), shuffles, and plays each exactly once.

**index.html**
```html
<!doctype html>
<html>
<head>
  <meta charset="utf-8" />
  <meta name="viewport" content="width=device-width, initial-scale=1" />
  <title>Folder Shuffle Player</title>
  <style>
    body { font-family: system-ui; padding: 20px; }
    button { padding: 10px 14px; margin-right: 8px; }
    #now { margin-top: 12px; }
    ol { margin-top: 10px; }
  </style>
</head>
<body>
  <h1>Folder Shuffle Player</h1>

  <input id="folder" type="file" webkitdirectory multiple />
  <button id="play" disabled>Play (shuffle)</button>
  <button id="stop">Stop</button>

  <div id="now">Now playing: —</div>
  <div id="status"></div>
  <ol id="list"></ol>

  <audio id="audio"></audio>

  <script>
    const folderInput = document.getElementById("folder");
    const playBtn = document.getElementById("play");
    const stopBtn = document.getElementById("stop");
    const audio = document.getElementById("audio");
    const now = document.getElementById("now");
    const status = document.getElementById("status");
    const listEl = document.getElementById("list");

    const AUDIO_EXT = /\.(mp3|wav|ogg|m4a|aac|flac)$/i;

    let files = [];          // File objects
    let urls = [];           // object URLs (for playback)
    let queue = [];          // shuffled indices
    let idx = -1;

    function shuffle(arr) {
      for (let i = arr.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [arr[i], arr[j]] = [arr[j], arr[i]];
      }
      return arr;
    }

    function cleanupObjectUrls() {
      urls.forEach(u => URL.revokeObjectURL(u));
      urls = [];
    }

    folderInput.addEventListener("change", () => {
      cleanupObjectUrls();

      const selected = Array.from(folderInput.files || []);
      files = selected
        .filter(f => AUDIO_EXT.test(f.name))
        .sort((a, b) => (a.webkitRelativePath || a.name).localeCompare(b.webkitRelativePath || b.name));

      urls = files.map(f => URL.createObjectURL(f));

      listEl.innerHTML = files.map(f => `<li>${f.webkitRelativePath || f.name}</li>`).join("");
      status.textContent = files.length ? `${files.length} audio files loaded.` : `No audio files found in that folder.`;

      now.textContent = "Now playing: —";

    function startRun() {
      if (!urls.length) return;
      queue = shuffle([...Array(urls.length).keys()]); // indices, no repeats
      idx = 0;
      playCurrent();

      status.textContent = "Playing…";
      audio.play();

        audio.removeAttribute("src");
        audio.load();
        now.textContent = "Now playing: —";
        status.textContent = "Done. Click Play to reshuffle and run again.";
        idx = -1;

      audio.pause();
      audio.currentTime = 0;
      audio.removeAttribute("src");
      audio.load();
      idx = -1;
      queue = [];
      now.textContent = "Now playing: —";
      status.textContent = "Stopped.";
    });

    audio.addEventListener("ended", next);
```
